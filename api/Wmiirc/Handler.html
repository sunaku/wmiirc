def handle key, *args, &block
  if block
    self[key] << block

  elsif key? key
    self[key].each do |block|
      begin
        block.call(*args)
      rescue LocalJumpError => error
        # let handlers return early, just like methods
        raise unless error.message == 'unexpected return'
      end
    end
  end

  block
end
